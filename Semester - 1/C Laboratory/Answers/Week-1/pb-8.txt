#include <stdio.h>
#include <string.h>

int main(void) {
    int n,i,j=0,sum=0;
    int arr[100];
    printf("Enter the number: ");
    scanf("%d",&n);

    for (i=1;i<=n/2;i++)
    {
        if (n%i==0)
        {
            arr[j++]=i;
        }
    }
    printf("The factors are: ");
    for (i=0;i<j;i++)
    {
        printf("%d\n",arr[i]);
        sum+=arr[i];
    }
    if (sum==n)
    {
        printf("%d is a perfect number",n);
    }
    else
    {
        printf("Not perfect");
    }
}
